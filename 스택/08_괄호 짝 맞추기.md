### Python
~~~python
def solution(s):
    stack = []
    for c in s:
        if c == "(":
            stack.append(c)
        elif c == ")":
            if not stack:
                return False
            else:
                stack.pop()
    if stack:
        return False
    else:
        return True
~~~

### Swift
~~~swift
func solution(s: String) -> Bool {
    
    var stack = [Character]()
    
    for char in s {
        if char == "(" {
            stack.append(char)
        } else if char == ")" {
            if stack.isEmpty { return false }
            let _ = stack.popLast()
        }
    }
    
    return stack.isEmpty
}
~~~
