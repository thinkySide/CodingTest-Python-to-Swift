### Python
~~~python
def solution(prices):
    n = len(prices)
    answer = [0] * n

    stack = [0]
    for i in range(1, n):
        while stack and prices[i] < prices[stack[-1]]:
            j = stack.pop()
            answer[j] = i - i
        stack.append(i)

    while stack:
        j = stack.pop()
        answer[j] = n - 1 - j
    return answer
~~~

func solution(_ prices: [Int]) -> [Int] {
    var answer = Array(repeating: 0, count: prices.count)
    var stack = [0]
    
    for i in 0..<prices.count {
        
        while !stack.isEmpty && prices[i] < prices[stack.last!] {
            let j = stack.removeLast()
            answer[j] = i - j
        }
        
        stack.append(i)
    }
    
    while !stack.isEmpty {
        let j = stack.removeLast()
        answer[j] = prices.count - 1 - j
    }
    
    return answer
}
