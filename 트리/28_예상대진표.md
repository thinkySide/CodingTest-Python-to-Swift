func solution(_ n:Int, _ a:Int, _ b:Int) -> Int {
    var answer = 0
    var copyA = a
    var copyB = b

    while copyA != copyB {
        if copyA % 2 == 0 { copyA /= 2 }
        else { copyA = (copyA + 1) / 2 }
        
        if copyB % 2 == 0 { copyB /= 2 }
        else { copyB = (copyB + 1) / 2 }
        
        answer += 1
    }

    return answer
}
